### Task 1: 添加 MenuState — ROM 扫描与菜单渲染

**Files:**
- Modify: `main.go`

- [ ] **Step 1: 添加 MenuState 结构体和方法**

在 `main.go` 中添加（`package main` 之后，`type Game struct` 之前）：

```go
import (
	"log"
	"os"
	"path/filepath"
	"sort"

	"fcProject/apu"
	"fcProject/bus"
	"fcProject/cartridge"
	"fcProject/cpu"
	"fcProject/input"
	"fcProject/ppu"

	"github.com/hajimehoshi/ebiten/v2"
	"github.com/hajimehoshi/ebiten/v2/text"
	"golang.org/x/image/font/basicfont"
)

type MenuState struct {
	romFiles   []string
	cursor     int
	face       basicfont.Face
}

func NewMenu() *MenuState {
	// Scan rom/ directory for .nes files
	entries, err := os.ReadDir("rom")
	if err != nil {
		log.Fatal(err)
	}
	var files []string
	for _, e := range entries {
		if !e.IsDir() && filepath.Ext(e.Name()) == ".nes" {
			files = append(files, e.Name())
		}
	}
	sort.Strings(files)
	return &MenuState{
		romFiles: files,
		face:     *basicfont.Face7x13,
	}
}

func (m *MenuState) Update() error {
	// Up/Down to move cursor
	if ebiten.IsKeyPressed(ebiten.KeyArrowUp) {
		if m.cursor > 0 { m.cursor-- }
	}
	if ebiten.IsKeyPressed(ebiten.KeyArrowDown) {
		if m.cursor < len(m.romFiles)-1 { m.cursor++ }
	}
	return nil
}

func (m *MenuState) Draw(screen *ebiten.Image) {
	screen.Fill(color.Black)
	// Title
	title := "Select Game"
	text.Draw(screen, title, m.face, 300, 200, color.White)
	// Game list
	for i, name := range m.romFiles {
		y := 240 + i*20
		c := color.White
		if i == m.cursor {
			c = color.Black
			// Draw highlight background
			highlight := ebiten.NewImage(200, 16)
			highlight.Fill(color.White)
			op := &ebiten.DrawImageOptions{}
			op.GeoM.Translate(300, float64(y-12))
			screen.DrawImage(highlight, op)
		}
		text.Draw(screen, name, m.face, 300, y, c)
	}
	// Footer
	text.Draw(screen, "Enter: Start  Up/Down: Select", m.face, 260, 600, color.RGBA{128, 128, 128, 255})
}

func (m *MenuState) Layout(w, h int) (int, int) { return screenWidth, screenHeight }

func (m *MenuState) Selected() string {
	if m.cursor < len(m.romFiles) {
		return m.romFiles[m.cursor]
	}
	return ""
}
```

### Task 2: 重构 Game 为 GameState + 添加 App 状态机

**Files:**
- Modify: `main.go`

- [ ] **Step 1: 将 `type Game struct` 重命名为 `type GameState struct`，保持所有方法不变**

`Game` → `GameState`，`NewGame` → `NewGameState`。`Update`/`Draw`/`Layout` 方法接收者从 `*Game` 改为 `*GameState`。

- [ ] **Step 2: 添加 App 结构体**

```go
type App struct {
	state    AppState
	menu     *MenuState
	game     *GameState
}

type AppState int

const (
	StateMenu AppState = iota
	StateGame
)

func NewApp() *App {
	return &App{
		state: StateMenu,
		menu:  NewMenu(),
	}
}

func (a *App) Update() error {
	switch a.state {
	case StateMenu:
		a.menu.Update()
		// Enter to start game
		if ebiten.IsKeyPressed(ebiten.KeyEnter) {
			sel := a.menu.Selected()
			if sel != "" {
				g, err := NewGameState("rom/" + sel)
				if err != nil {
					return err
				}
				a.game = g
				a.state = StateGame
			}
		}
	case StateGame:
		a.game.Update()
		// ESC to return to menu
		if ebiten.IsKeyPressed(ebiten.KeyEscape) {
			a.state = StateMenu
		}
	}
	return nil
}

func (a *App) Draw(screen *ebiten.Image) {
	switch a.state {
	case StateMenu:
		a.menu.Draw(screen)
	case StateGame:
		a.game.Draw(screen)
	}
}

func (a *App) Layout(w, h int) (int, int) {
	return screenWidth, screenHeight
}
```

### Task 3: 修复按键重复触发问题 + 验证

**Files:**
- Modify: `main.go`

- [ ] **Step 1: 添加按键去抖动**

MenuState 和 App 中的 `IsKeyPressed` 会导致每一帧都触发（60fps → 光标飞移）。需要记录上一帧按键状态，只在"未按→按下"时触发。

在 `App` struct 添加 `prevEnter bool`、`prevESC bool`。在 `MenuState` struct 添加 `prevUp bool`、`prevDown bool`。

修改 Update 方法中的按键检测为"边缘触发"：

```go
// In MenuState.Update():
upPressed := ebiten.IsKeyPressed(ebiten.KeyArrowUp)
downPressed := ebiten.IsKeyPressed(ebiten.KeyArrowDown)
if upPressed && !m.prevUp && m.cursor > 0 { m.cursor-- }
if downPressed && !m.prevDown && m.cursor < len(m.romFiles)-1 { m.cursor++ }
m.prevUp = upPressed
m.prevDown = downPressed

// In App.Update():
enterPressed := ebiten.IsKeyPressed(ebiten.KeyEnter)
escPressed := ebiten.IsKeyPressed(ebiten.KeyEscape)
if enterPressed && !a.prevEnter { /* start game */ }
if escPressed && !a.prevESC { /* return menu */ }
a.prevEnter = enterPressed
a.prevESC = escPressed
```
